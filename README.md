## set environment variable
   - export YII_ENV="dev"
   - export DB_DSN="mysql:host=your db host;dbname=your db name"
   - export DB_USERNAME="your db username"
   - export DB_PASSWORD="your db pwd"
## install
   - git clone
   - composer install
   - php yii migrate
## run
   - php yii serve
